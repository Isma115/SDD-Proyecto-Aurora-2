```javascript
const { Sequelize } = require('sequelize'); // Importamos la librería sequelize

// Creamos el modelo de cliente
const Cliente = sequelize.define('Cliente', {
  nombre: { type: Sequelize.STRING, allowNull: false },
  apellido: { type: Sequelize.STRING, allowNull: false },
  email: {
    type: Sequelize.STRING,
    allowNull: false,
    unique: true, // Asegurar que el email sea único
    validate: {
      isEmail: true,
    },
  },
  telefono: { type: Sequelize.STRING, allowNull: true },
  fechaRegistro: {
    type: Sequelize.DATE,
    defaultValue: Sequelize.NOW, // Define la fecha de creación por defecto
  }
});

module.exports = Cliente; 

```

    *  `email`: Email del cliente (STRING, no puede estar vacío).
